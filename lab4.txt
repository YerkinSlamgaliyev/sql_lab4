SELECT AVG(Возраст) AS Средний_возраст FROM users;

SELECT Модель, AVG(Время_за_экраном) AS Среднее_время
FROM users
GROUP BY Модель;

SELECT ОС, SUM(Использование_данных) AS Общий_объем_данных
FROM users
GROUP BY ОС;

SELECT Модель, COUNT(*) AS Количество_пользователей
FROM users
GROUP BY Модель;

SELECT Пол, AVG(Время_за_экраном) AS Среднее_время, AVG(Использование_данных) AS Среднее_использование
FROM users
GROUP BY Пол;

SELECT * 
FROM users
WHERE Возраст < 30;

SELECT ID, Модель, Время_за_экраном 
FROM users
WHERE Время_за_экраном > 400;

SELECT MAX(Использование_данных) AS Максимальное_использование
FROM users
WHERE ОС = 'iOS';

SELECT MIN(Возраст) AS Минимальный_возраст
FROM users
WHERE Модель = 'Google Pixel 5';

SELECT Возраст, SUM(Время_за_экраном) AS Общее_время
FROM users
GROUP BY Возраст;

SELECT Модель, AVG(Использование_данных) AS Среднее_использование
FROM users
WHERE ОС = 'Android'
GROUP BY Модель;

SELECT Пол, COUNT(*) AS Количество
FROM users
WHERE ОС = 'Android'
GROUP BY Пол;

SELECT *
FROM users
WHERE Время_за_экраном = (SELECT MIN(Время_за_экраном) FROM users);
		
SELECT 
    CASE 
        WHEN Возраст < 30 THEN 'Моложе 30'
        WHEN Возраст BETWEEN 30 AND 40 THEN '30-40'
        ELSE 'Старше 40'
    END AS Возрастная_группа,
    SUM(Использование_данных) AS Общий_объем_данных
FROM users
GROUP BY Возрастная_группа;


SELECT ID, Модель, Использование_данных
FROM users
ORDER BY Использование_данных DESC
LIMIT 3;
